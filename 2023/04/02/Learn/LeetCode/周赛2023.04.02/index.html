最长平衡子字符串

思路

使用双指针并用num计000111的数。虽然这样做空间复杂度能做到O(n)，但是相对的产生的判断很多。

考虑的情况太多，不易快速解决。

My Code

class Solution {
public:
    int findTheLongestBalancedSubstring(string s) {
        if(s.size()<=1){return 0;}
        int left = 0;
        int right = 0;
        int last_right = 0;
        int res = 0;
        int num = 0;
        bool flag = false;
        for(int i=0;i<s.size();i++){
            if(s[right]=='0'){
                num++;
                flag=true;
            }
            if(s[right]=='1'){num--;}
            if(num<0){
                left=right;
                num=0;
            }
            //cout<<num;
            if(num==0 && flag==true){
                res = max(res,right-left+1);
                flag = false;
                //cout<<flag<<i<<' ';
            }
            else if(i==s.size()-1 && flag==true && num>0 && s[i]=='1'){
                res = max(res,right-left+1-num);
                //cout<<' '<<num;
                //cout<<right<<' '<<left;
                flag = false;
            }
            if(s[right]=='0' && s[last_right]=='1'){
                res = max(res,right-left+1-num);
                left = right;
                num = 1;
            }
            cout<<left<<' '<<right<<' ';
            last_right = right;
            right++;
        }
        if(res%2!=0)res=0;
        return res;
    }
};

不足

考虑的方法不是很好，导致一直在debug，解决时间花了很久，并且好几次提交错误。

先想解决问题再去减少复杂度。

学习

找到1然后再去找1前面的0！逆天。我太菜了。

举例0011，先找到1后看1前面的0，cur=2；继续找1，发现011再去找0前面是不是0，如果是就0011，cur=4，记录cur最大值。

class Solution {
public:
    int findTheLongestBalancedSubstring(string s) {
        int n = s.size(), ret = 0, cur = 0;
        for(int i = 0; i < n; ++i){
            if(s[i] == '1' && i - cur - 1 >= 0 && s[i - cur - 1] == '0')
                cur += 2;
            else
                cur = 0;
            ret = max(ret, cur);
        }
        return ret;
    }
};

转换二位数组

思路

考虑了递归，从重复的地方开始往后遍历，记录在map里，并通过find查找map里有没有重复。

使用过的数置为0，便于下次遍历。

My Code

class Solution {
public:
    vector<int> path;
    vector<vector<int>> result;
    map<int,int> mp;
    int count = 0;
    bool flag = 0;
    int start = 0;
    void backtracing(vector<int>& nums, int startIndex){
        if(count==nums.size()){return;}
        for(int i=startIndex;i<nums.size();i++){
            cout<<i<<' ';
            if(mp.find(nums[i])==mp.end() && nums[i]!=0){

                mp[nums[i]]++;
                count++;
                path.push_back(nums[i]);
                nums[i]=0;
            }
            else if(flag == 0 && mp.find(nums[i])!=mp.end()){
                start = i;
                flag = 1;
                //cout<<i;
            }
            if(i==nums.size()-1){
                result.push_back(path);
                path.clear();
                mp.clear();
                flag = 0;
                backtracing(nums, start);
                return;
            }
        }
        return;
    }
    vector<vector<int>> findMatrix(vector<int>& nums) {
        if(nums.size()==0){return result;}
        backtracing(nums,0);
        return result;
    }
};

不足

应该把全部先放进map，再在map中减，这样会方便很多。遍历map，如果value还有值就留在map，如果没有值了就从map中删除，删除用erase。

class Solution {
public:
    vector<vector<int>> findMatrix(vector<int>& nums) {
        map<int, int> mp;
        for (int x : nums) {
            mp[x] += 1;
        }
        vector<vector<int>> ans;
        while (not mp.empty()) {
            vector<int> res;
            for (auto it = mp.begin(); it != mp.end(); ) {
                res.push_back(it->first);
                if (it->second -= 1) {
                    it = next(it);
                } else {
                    it = mp.erase(it);
                }
            }
            ans.push_back(res);
        }
        return ans;
    }
};

老鼠和奶酪

思路

这题最重要的是思路，先对(r1-r2)进行排序，取最大的k个r1[0-k]，剩下的都拿r2[k-size]。那么如何计算最大得分？

我们在排序的过程中已经计算出来前k个的r1-r2[0-k]，那么我们这时候再加上全部r2[0-size]，就是问题的解了。

可以看到加起来是r1[0-k]+r2[k-size]。

My Code

class Solution {
public:
    int miceAndCheese(vector<int>& reward1, vector<int>& reward2, int k) {
        int ans = 0;
        int n = reward1.size();
        vector<int> v;
        for (int i = 0; i < n; i += 1) {
            ans += reward2[i];
            v.push_back(reward1[i] - reward2[i]);
        }
        sort(v.begin(), v.end(), greater<int>());
        for (int i = 0; i < k; i += 1) {
            ans += v[i];
        }
        return ans;
    }
};

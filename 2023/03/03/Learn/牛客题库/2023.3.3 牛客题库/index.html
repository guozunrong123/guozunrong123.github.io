1、字符串最后一个单词的长度

My Code

#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <string>
using namespace std;

int main() {
    string str;
    int len = 0;
    getline(cin,str);
    for(int i=0; i<str.length(); i++){
        if(str[i] == ' '){
            len = 0;        
        }
        else{
            len++;
            }
    }
    cout << len;
    return 0;
}

Learn

  1. 包含头文件#include

    定义字符串String str;

    字符串长度str.length()

  2. int 后面一定要赋值！！

  3. cin 遇空格或换行，会停止识别，如果你打算输入的字符串中带1个或多个空格，则采用getline把停止识别的符号设置为‘n’（即换行符），就能正确输入输出了。

    例：getline(cin,str);

  4. 他人解法：

  • while(cin >> a)，使用一个istream的对象作为条件时，效果是检查流的状态。若流有效，则检测成功，返回true。当遇到文件结束符或者无效的输入（比如用一个字符来作为一个整型数的输入时），istream的对象状态会变为无效。
  • size和length这两个函数会返回 string 类型对象中的字符个数，且它们的执行效果相同。
#include <iostream>
using namespace std;
#include <string>

int main() {
    string a;
    int b;
    while (cin >> a ) { // 注意 while 处理多个 case
        b = a.size();
    }
    cout << b << endl;
    return 0;
}
// 64 位输出请用 printf("%lld")

2、计算某字符

My Code

#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    char str_check = {0};
    int amount = 0;
    getline(cin,str);
    cin >> str_check;
    for(int i = 0;i< str.length();i++){
        if(str[i] == str_check ){
            amount++;
        }
        else if(str_check >= 0x41 && str_check < 0x61){
            if(str[i] == str_check + 0x20 ){
                amount++;
            }
        }
        else if(str_check >= 0x61 && str_check < 0x81){
            if(str[i] == str_check - 0x20 ){
                amount++;
            }
        }        
    }
    cout << amount <<endl;
}

Learn

tolower(c);

函数说明：若参数 c 为大写字母则将该对应的小写字母返回。

返回值：返回转换后的小写字母，若不须转换则将参数c 值返回。

3、明明的随机数

My Code

#include <iostream>
using namespace std;

int main() {
    int num = 0;
    int randnum[1000] = {0};
    int nrepnum[500] = {0};
    int k = 0;
    bool is_same;
    cin >> num;
    for(int i=0; i < num; i++)
    {
        cin >> randnum[i];
    }
    for(int j=0; j < num; j++)
    {
        for(int a=0; a < k+1; a++)
        {
            if(nrepnum[a] == randnum[j])
            {
                is_same = true;
                break;
            }
        }
        if(is_same == true)
        {
            is_same = false;
        }
        else{
            nrepnum[k] = randnum[j];
            k++;
        }
    }
    for(int g=0; g<k-1; g++)
    {
        for(int h=0 ; h<k-1-g; h++){
            if(nrepnum[h] > nrepnum[h+1]){
                int temp = 0;
                temp = nrepnum[h];
                nrepnum[h] = nrepnum[h+1];
                nrepnum[h+1] = temp;
            }
        }
    }
    for(int b=0; b<k; b++)
    {
        cout << nrepnum[b] <<endl;
    }
}

Learn

有点牛逼…

思路：取最大数组，循环遍历所有值

将该值放入数组对应该位，使该位为true，这个操作直接删去了重复值，而且排好了序！！

循环遍历数组，判断数组对应位是否为true，是则输出。

#include <iostream>
using namespace std;
int main() {
    int N, n;
    while (cin >> N) {
        int a[1001] = { 0 };
        while (N--) {
            cin >> n;
            a[n] = 1;
        }
        for (int i = 0; i < 1001; i++)
            if (a[i])
                cout << i << endl;
    }
    return 0;
}

Learn

1.获取子字符串substr()

  • 函数原型: string substr(size_t pos = 0, size_t len = npos) const;

  • 功能: 从子字符串中获取想要的子字符串

  • 参数：

    • pos: 想要获取的子字符串的第一个字符的位置，如果pos等于字符串长度，则该函数返回一个空字符串，如果等于字符串长度，则该函数返回一个空的字符串，如果该长度大于字符串长度，则抛出一个out_of_range。 注：第一个字符的下标从0开始。
    • len： 子字符串中要包含的字符数, string::npos的值表示知道字符串末尾的所有字符。

  • 返回值：带有对象子字符串的字符串对象。

2.insert()

basic_string& insert (size_type pos, const basic_string& str);

在原串下标为pos的字符前插入字符串str

basic_string& insert (size_type pos, const basic_string& str, size_type pos1, size_type n);

str从下标为pos1开始数的n个字符插在原串下标为pos的字符前

basic_string& insert (size_type pos, size_type n, char c);

在原串下标为pos的字符前插入n个字符c

#include <string>
#include <vector>
#include <iostream>
using namespace std;

int main() {
    string s;
    getline(cin, s);
    while (true) {
        int len = s.size();
        if (len <= 8) {
            s.insert(s.end(), 8-len, '0');
            cout << s << endl;
            if (!getline(cin, s)) break;
        }
        else {
            cout << s.substr(0, 8) << endl;
            s = s.substr(8, len-8);
        }
    }
    return 0;
}

5、进制转换

My Code

#include <iostream>
#include <string>
#include <math.h>
using namespace std;

int main() {
    string hex;
    long int dec = 0;
    getline(cin,hex);
    int length = hex.length();
    for(int i=0; i<length-2 ; i++){
        if(hex[length-i-1] >= 'A' && hex[length-i-1] <= 'F')
        {
            hex[length-i-1] = hex[length-i-1] - 55;
        }
        else
        {
            hex[length-i-1] = hex[length-i-1] - 48;
        }
        dec += pow(16,i)*hex[length-i-1];
    }
    cout << dec << endl;
    return 0;
}

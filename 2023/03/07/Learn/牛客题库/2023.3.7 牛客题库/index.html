1、简单密码

My Code

#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    getline(cin,str);
    for(int i=0;i<str.length();i++){
        if(str[i] >= 'A' && str[i] < 'Z')
        {
            str[i] = str[i] - ('A' - 'a') + 1;
        }
        else if(str[i] == 'Z')
        {
            str[i] = 'a';
        }
        else if(str[i] >= 'a' && str[i]<='o'){
            str[i] = ((str[i]-1)/3 - 30) + '0';
        }
        else if(str[i] >= 'p' && str[i]<='s'){
            str[i] = '7';
        }
        else if(str[i] >= 't' && str[i]<='v'){
            str[i] = '8';
        }
        else if(str[i] >= 'w' && str[i]<='z'){
            str[i] = '9';
        }
    }
    cout << str ;
    
}

2、汽水瓶

My Code

#include <iostream>
using namespace std;

int count = 0;
int bottleChange(int k){
    while(k != 0){
        if(k<2){
            k = 0; //如果小于两个空瓶子，换不了
        }
        else if(k == 2){
            count ++;
            k = 0;//如果有两个空瓶子，可以换一个
        }
        else if( k >= 3){
            count += k/3;
            k = (k/3)+(k%3);//如果有三个以上的空瓶子，先换在加没换的，再循环
        }
    }
    return 0;
}

int main() {
    int n = 0;
    while(cin >> n) {
        if(n==0)
        {
            return 0;
        }
        bottleChange(n);
        cout << count <<endl;
        count = 0;
    }
    return 0;
}

3、删除字符串中出现次数最少的字符

My Code

#include <iostream>
#include <vector>
using namespace std;

int main()
{
    string str;
    
    while(cin>>str)
    {
        int min=20;
        vector<int> Hash(26,0);
        for(int i=0;i<str.size();i++)
        {
            Hash[str[i]-'a']++;
        }
        for(int i=0;i<26;i++)
        {
            if(min>Hash[i]&&Hash[i]>0)
                min=Hash[i];
        }
        for(int i=0;i<str.size();i++)
        {
            if(Hash[str[i]-'a']!=min)
                cout<<str[i];
        }
        cout<<endl;
    }
    return 0;
}

4、合唱队

My Code

#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int str[n];
    for(int i=0;i<n;i++){
        cin >> str[i];
    }
    int left[n];
    int right[n];
    int left_min = 0;
    int left_count = 0;
    int right_min = 0;
    int right_count = 0; 
    int count = n;
    for(int i=0,j=n-i-1;i<n;i++,j--){
        left[i]=1;
        for(int ii=0;ii<i;ii++){
            if(str[i]>str[ii])
            {
                left[i] = max(left[i],left[ii]+1);
            }
        }
        right[j]=1;
        for(int jj=n-1;jj>j;jj--)
        {
            if(str[j]>str[jj]){
                right[j] = max(right[j],right[jj]+1);
            }
        }
    }
    for(int i=0;i<n;i++){
        count = min(count,n-(left[i]+right[i]-1));
    }

    cout << count;
}

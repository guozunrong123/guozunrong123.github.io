1、对角线上的质数

思路

判断斜线是否为质数并统计最大值

重点在判断是否为质数

Code

class Solution {
public:
    bool istrue(int nums){
        if(nums==1)return false;
        else if(nums<=3)return true;
        else if(nums%2==0)return false;
        for(int i=3;i*i<=nums;i+=2){
            if(nums%i==0)return false;
        }
        return true;
    }
    int maxRes=0;
    int diagonalPrime(vector<vector<int>>& nums) {
        for(int i=0;i<nums.size();i++){
            if(istrue(nums[i][i])){
                maxRes=max(maxRes, nums[i][i]);               
            }
            if(istrue(nums[i][nums.size()-i-1])){
                maxRes=max(maxRes, nums[i][nums.size()-i-1]); 
            }
        }
    return maxRes;
    }
};

2、等值距离和

思路

暴力解法：略，超时，时间复杂度O(n²)

fan_of_emptyhope解法：两次for，第一次从前往后遍历，用map记录相同的值的count和其下标之和，res = count * i - sum。第二次从后往前遍历，同上。两次res相加即为答案。时间复杂度O(n)

Code

第一次提交，超时。超暴力解法。

class Solution {
public:
    vector<long long> distance(vector<int>& nums) {
        vector<long long> res(nums.size(),0);
        for(long long i=0;i<nums.size();i++){            
            for(long long j=i;j<nums.size();j++){
                if(nums[i]==nums[j] && i!=j){
                    res[i]+=abs(i-j);
                    res[j]+=abs(i-j);
                }        
            }
        }
        return res;
    }
};

class Solution {
public:
    vector<long long> distance(vector<int>& nums) {
        map<int,int> count;
        map<int,long long> sum;
        vector<long long> res(nums.size(),0);
        for(int i=0;i<nums.size();i++){
            res[i] = 1ll*i*count[nums[i]] - sum[nums[i]];
            count[nums[i]]++;
            sum[nums[i]] += i;
        }
        count.clear();
        sum.clear();
        for(int i=nums.size()-1;i>=0;i--){
            res[i] +=  sum[nums[i]] - 1ll*i*count[nums[i]];
            count[nums[i]]++;
            sum[nums[i]] += i;
        }
        return res;
    }
};

3、最小化数对的最大差值

思路

二分答案。

Code

class Solution {
public:
    int minimizeMax(vector<int> &nums, int p) {
        sort(nums.begin(), nums.end());
        int left = -1, right = nums.back() - nums[0]; // 开区间
        while (left + 1 < right) { // 开区间
            int mid = left + (right - left) / 2, cnt = 0;
            for (int i = 0; i < nums.size() - 1; ++i)
                if (nums[i + 1] - nums[i] <= mid) { // 都选
                    ++cnt;
                    ++i;
                }
            (cnt >= p ? right : left) = mid;
        }
        return right;
    }
};

单调栈

class Solution {
public:
    int minimumVisitedCells(vector<vector<int>> &grid) {
        int m = grid.size(), n = grid[0].size(), mn;
        vector<vector<pair<int, int>>> col_st(n); // 每列的单调栈
        for (int i = m - 1; i >= 0; --i) {
            vector<pair<int, int>> st; // 当前行的单调栈
            for (int j = n - 1; j >= 0; --j) {
                auto &st2 = col_st[j];
                mn = INT_MAX;
                if (i == m - 1 && j == n - 1) // 特殊情况：已经是终点
                    mn = 0;
                else if (int g = grid[i][j]; g) {
                    // 在单调栈上二分
                    auto it = lower_bound(st.begin(), st.end(), j + g, [](const auto &a, const int b) {
                        return a.second > b;
                    });
                    if (it < st.end()) mn = min(mn, it->first);
                    it = lower_bound(st2.begin(), st2.end(), i + g, [](const auto &a, const int b) {
                        return a.second > b;
                    });
                    if (it < st2.end()) mn = min(mn, it->first);
                }
                if (mn == INT_MAX) continue;

                ++mn; // 加上 (i,j) 这个格子
                // 插入单调栈
                while (!st.empty() && mn <= st.back().first)
                    st.pop_back();
                st.emplace_back(mn, j);
                while (!st2.empty() && mn <= st2.back().first)
                    st2.pop_back();
                st2.emplace_back(mn, i);
            }
        }
        return mn < INT_MAX ? mn : -1;
    }
};

1、质数因子

这题卡了好久，一直有一个案例不通过。原因是忽略了质数的质数是它本身！考虑的不够全面。

My Code

#include <iostream>
#include <math.h>
using namespace std;

int main() {
    int numIn = 0;
    cin >> numIn;
    for(int i=2; i<=sqrt(numIn); i++)
    {
        while(numIn % i == 0)
        {
            cout<< i << ' ';
            numIn = numIn/i;
        }
    }
    if(numIn!=1)
    {
        cout<< numIn << ' ';
    }
    return 0;
}

2、取近似值

My Code

思路：浮点转整型后面的小数会去掉，那如果浮点乘以10减去转型的乘以10，那么就得到了小数点后一位的数字。

#include <iostream>
using namespace std;

int main() {
    double numIn = 0;
    long int num = 0;
    cin >> numIn;
    long int numtemp = (numIn*10);
    numtemp -= (int)numIn*10;
    if(numtemp < 5){
        numtemp = 0;
    }
    else {
        numtemp = 1;
    }
    num = (int)numIn + numtemp;
    cout << num;
}

Learn

乘以10%10就可以得到了….好像是。。晕

4、提取不重复的整数

My Code

#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    getline(cin,str);
    for(int i=0;i<str.length()/2;i++){
        char temp = str[i];
        str[i] = str[str.length()-1-i];
        str[str.length()-1-i] = temp;
    }
    for(int i=0;i<str.length();i++){
        for(int j=i+1;j<str.length();j++){
            if(str[i] == str[j])
            {
                str[j]=-1;
            }
        }
    }
    for(int i=0;i<str.length();i++){
        if(str[i]!=-1)
        {
        cout << str[i];
        }
    }

}

5、字符个数统计

My Code

思路：先将重复的字符置为’ ‘，再计算不是’ ‘的数量

#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    getline(cin,str);
    int Num = 0;
    for(int i=0;i<str.length();i++)
    {
        for(int j=i+1;j<str.length();j++)
        {
            if(str[i] == str[j] && str[i] != ' ')
            {
                str[j] = ' ';
            }
        }
        if(str[i] != ' '){
            Num++;
        }
    }
    cout << Num;
}

Learn

采用set容器，使用set的insert去消除重复项，最后输出set的大小

#include <iostream>
#include <set>
using namespace std;
int main()
{
    char s;
    set<char> a;
    while(cin>>s)
    {
        a.insert(s);
    }
    cout<<a.size();
}
